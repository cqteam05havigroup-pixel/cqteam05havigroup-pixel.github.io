function calcLo(){

  const seaR = +document.getElementById('l-sea').value||2;
  const dom  = +document.getElementById('l-dom').value||5.5;

  const sell = +document.getElementById('l-sell').value||0;
  const wt   = +document.getElementById('l-weight').value||0;
  const moq  = +document.getElementById('l-moq').value||100;
  const adsP = +document.getElementById('l-ads').value||0;

  const rkg     = shipMode==='sea'?seaR:airR;
  const cpShip  = wt*rkg;
  const cogs    = cost+cpShip;

function calcScore(){
  const sold  = +document.getElementById('r-sold').value||0;
  const watch = +document.getElementById('r-watch').value||0;
  const price = +document.getElementById('r-price').value||0;
  const comp  = +document.getElementById('r-comp').value||0;
  const vero  = +document.getElementById('r-vero').value;
  const trend = +document.getElementById('r-trend').value;

  let score=0;
  // Sold score (0-30)
  score += Math.min(sold/5,30);
  // Watch score (0-20)
  const watchRate = sold>0?watch/sold*100:0;
  score += Math.min(watchRate/2,20);
  // Price sweet spot $10-50 (0-20)
  score += price>=10&&price<=50?20:price>5&&price<80?12:5;
  // Competition (0-15) - less is more
  score += comp<=5?15:comp<=15?10:comp<=30?5:0;
  // VERO (0-10)
  score += vero===0?10:vero===1?5:0;
  // Trend (0-15)
  score += trend===2?15:trend===1?8:0;

  score = Math.min(Math.round(score),100);

  document.getElementById('score-num').textContent = score;
  const ring=document.getElementById('ring-fill');
  const offset=283-(283*score/100);
  ring.style.strokeDashoffset=offset;
  ring.style.stroke=score>=70?'var(--green)':score>=50?'var(--yellow)':'var(--red)';

  let verd,sub;
  if(score>=75){verd='🔥 RẤT TIỀM NĂNG';sub='Nên research sâu và sourcing ngay';}
  else if(score>=55){verd='✅ KHÁ TỐT';sub='Có thể test với lô nhỏ';}
  else if(score>=35){verd='⚠ TRUNG BÌNH';sub='Cần optimize thêm hoặc tìm niche khác';}
  else{verd='❌ KHÔNG NÊN';sub='Rủi ro cao, không phù hợp';}
  document.getElementById('score-verdict').textContent=verd;
  document.getElementById('score-sub').textContent=sub;

  document.getElementById('sig-sold').textContent=sold+' đơn/tháng';
  document.getElementById('sig-watch').textContent=watchRate.toFixed(1)+'%';
  document.getElementById('sig-price').textContent='$'+price;
  document.getElementById('sig-comp').textContent=comp+' sellers';

  document.querySelectorAll('.signal').forEach(s=>s.className='signal neu');
  const sigs=document.querySelectorAll('.signal');
  if(sold>=50) sigs[0].className='signal pos'; else if(sold<20) sigs[0].className='signal neg';
  if(watchRate>=20) sigs[1].className='signal pos'; else if(watchRate<10) sigs[1].className='signal neg';
  if(price>=10&&price<=50) sigs[2].className='signal pos'; else if(price<5) sigs[2].className='signal neg';
  if(comp<=10) sigs[3].className='signal pos'; else if(comp>30) sigs[3].className='signal neg';

  document.getElementById('research-analysis').innerHTML = `
<strong style="color:var(--white)">📊 Phân tích sản phẩm (Score: ${score}/100)</strong><br><br>
${sold>=50?'✅':'⚠️'} <strong>Velocity:</strong> ${sold} đơn/tháng — ${sold>=100?'Demand rất tốt, market đã proven':'Demand ổn, cần test thêm'}<br>
${watchRate>=15?'✅':'⚠️'} <strong>Watch Rate:</strong> ${watchRate.toFixed(1)}% — ${watchRate>=20?'Người mua quan tâm cao':watchRate>=10?'Bình thường':'Thấp, cần cải thiện listing/photos'}<br>
${price>=10&&price<=50?'✅':'⚠️'} <strong>Price Point:</strong> $${price} — ${price>=15&&price<=45?'Sweet spot tốt cho eBay drop':'Cân nhắc lại pricing strategy'}<br>
${comp<=15?'✅':'⚠️'} <strong>Competition:</strong> ${comp} sellers — ${comp<=10?'Blue ocean, ít cạnh tranh':comp<=20?'Cạnh tranh vừa phải':'Đỏ, cần differentiation mạnh'}<br>
${vero===0?'✅':'❌'} <strong>VERO Risk:</strong> ${vero===0?'Safe — Không có risk bản quyền':vero===1?'Cần check thêm trước khi list':'CÓ RISK — Không nên proceed'}<br>

// ===== TITLE GENERATOR =====
function genTitle(){
  const name  = document.getElementById('t-name').value;
  const qty   = document.getElementById('t-qty').value;
  const mat   = document.getElementById('t-mat').value;
  const feat  = document.getElementById('t-feat').value;
  const use   = document.getElementById('t-use').value;
  const brand = document.getElementById('t-brand').value;

  if(!name){
    document.getElementById('title-output').textContent='Nhập tên sản phẩm để generate...';
    document.getElementById('char-count').textContent='0 / 80 ký tự';
    document.getElementById('title-variants').innerHTML='';
    return;
  }

  const parts=[name,qty,mat,feat,use,brand].filter(Boolean);
  const base=parts.join(' ');
  const title=base.substring(0,80).toUpperCase();
  const count=title.length;

  document.getElementById('title-output').textContent=title;
  const cc=document.getElementById('char-count');
  cc.textContent=count+' / 80 ký tự';
  cc.className='char-counter'+(count>80?' over':count>70?' warn':'');

  // Variants
  const variants=[
    [name,mat,feat,qty,use,brand].filter(Boolean).join(' ').substring(0,80).toUpperCase(),
    [feat,name,qty,mat,use].filter(Boolean).join(' ').substring(0,80).toUpperCase(),
    [qty,name,feat,mat,use,brand].filter(Boolean).join(' ').substring(0,80).toUpperCase(),
  ].filter((v,i,arr)=>v!==title&&arr.indexOf(v)===i);

  document.getElementById('title-variants').innerHTML=variants.map((v,i)=>`
    <div style="background:var(--s2);border:1px solid var(--border);border-radius:8px;padding:12px;font-family:var(--mono);font-size:12px;color:var(--muted);cursor:pointer;" 
         onclick="this.style.borderColor='var(--green)';document.getElementById('title-output').textContent=this.querySelector('.vt').textContent;">
      <div style="font-size:9px;color:var(--faint);letter-spacing:.1em;margin-bottom:4px;">VARIANT ${i+2} · ${v.length}/80</div>
      <div class="vt" style="color:var(--white)">${v}</div>
    </div>
  `).join('');
}
